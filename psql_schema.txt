CREATE TABLE Users (ID VARCHAR(1700) NOT NULL PRIMARY KEY, username VARCHAR(30), realname VARCHAR(30), avatar bytea);
CREATE TABLE Followers (follower VARCHAR(1700) NOT NULL REFERENCES Users(ID), poster VARCHAR(1700) NOT NULL REFERENCES Users(ID)) ;
CREATE TABLE Posts (postID SERIAL NOT NULL PRIMARY KEY, tags VARCHAR(30) ARRAY ,ID VARCHAR(1700), FOREIGN KEY(ID) REFERENCES Users(ID),postContent VARCHAR(500)) ;
CREATE TABLE Stocks(Ticker VARCHAR(20) PRIMARY KEY NOT NULL, name VARCHAR(20) NOT NULL, yearstart INTEGER);
CREATE TABLE Postmeta(PID SERIAL NOT NULL REFERENCES posts(postid), ticker VARCHAR(20) REFERENCES stocks(ticker));
CREATE TABLE Subscriptions(UID VARCHAR(1700) NOT NULL REFERENCES users(ID), Ticker VARCHAR(20) NOT NULL REFERENCES stocks(ticker));
